INHERITING THE __init__() Function
class Person:
    def __init__(self, fname, lname):
        self.fname = fname
        self.lname = lname

class Student(Person):
    def __init__(self, fname, lname, age):
        self.age = age
        Person.__init__(self, fname, lname) """Super().__init__(fname, lname)"""
